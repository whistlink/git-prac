package com.collections.javaswing;

import javax.swing.*;
import java.awt.*;
import java.awt.event.KeyAdapter;
import java.awt.event.KeyEvent;

public class RectangleCalculator extends JFrame {
    private final JTextField heightField;
    private final JTextField widthField;
    private final JLabel areaLabel;
    private final JLabel perimeterLabel;

    public RectangleCalculator() {
   
        setTitle("Rectangle Calculator");
        setSize(600, 300);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLayout(new GridLayout(4, 2));
        setResizable(false);
      
        add(new JLabel("Height:"));
        heightField = new JTextField();
        add(heightField);

        add(new JLabel("Width:"));
        widthField = new JTextField();
        add(widthField);

        areaLabel = new JLabel("Area: ");
        add(areaLabel);

        perimeterLabel = new JLabel("Perimeter: ");
        add(perimeterLabel);

        
        KeyAdapter updateListener = new KeyAdapter() {
            @Override
            public void keyReleased(KeyEvent e) {
                updateCalculations();
            }
        };

        heightField.addKeyListener(updateListener);
        widthField.addKeyListener(updateListener);
    }

    private void updateCalculations() {
        try {
            double height = Double.parseDouble(heightField.getText());
            double width = Double.parseDouble(widthField.getText());

            double area = height * width;
            double perimeter = 2 * (height + width);

            areaLabel.setText("Area: " + area);
            perimeterLabel.setText("Perimeter: " + perimeter);
        } catch (NumberFormatException e) {
            areaLabel.setText("Area: ");
            perimeterLabel.setText("Perimeter: ");
        }
    }
}

//radio buttons
package com.collections.javaswingrb;

import java.awt.GridLayout;
import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class JavaSwingRB extends JFrame {
    private final JTextField displayField;
    private final JRadioButton radioButton1;
    private final JRadioButton radioButton2;
    private final JRadioButton radioButton3;
    private final JRadioButton radioButton4;
    private final JRadioButton radioButton5;
    private final JCheckBox checkBox1;
    private final JCheckBox checkBox2;
    private final JCheckBox checkBox3;
    private final JCheckBox checkBox4;
    private final JCheckBox checkBox5;
    private final ButtonGroup radioGroup;

    public JavaSwingRB() {
        // Set up the frame
        setTitle("Radio and CheckBox");
        setSize(300, 250);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLayout(new GridLayout(7, 2));

        // Initialize display field
        displayField = new JTextField();
        displayField.setEditable(false);
        add(new JLabel("Selected:"));
        add(displayField);

        // Initialize radio buttons
        radioButton1 = new JRadioButton("1");
        radioButton2 = new JRadioButton("2");
        radioButton3 = new JRadioButton("3");
        radioButton4 = new JRadioButton("4");
        radioButton5 = new JRadioButton("5");
        radioGroup = new ButtonGroup();
        radioGroup.add(radioButton1);
        radioGroup.add(radioButton2);
        radioGroup.add(radioButton3);
        radioGroup.add(radioButton4);
        radioGroup.add(radioButton5);

        // Add radio buttons with ActionListener
        radioButton1.addActionListener(new RadioButtonListener());
        radioButton2.addActionListener(new RadioButtonListener());
        radioButton3.addActionListener(new RadioButtonListener());
        radioButton4.addActionListener(new RadioButtonListener());
        radioButton5.addActionListener(new RadioButtonListener());
        
        add(radioButton1);
        add(radioButton2);
        add(radioButton3);
        add(radioButton4);
        add(radioButton5);

        // Initialize checkboxes
        checkBox1 = new JCheckBox("1");
        checkBox2 = new JCheckBox("2");
        checkBox3 = new JCheckBox("3");
        checkBox4 = new JCheckBox("4");
        checkBox5 = new JCheckBox("5");

        // Add checkboxes with ActionListener
        checkBox1.addActionListener(new CheckBoxListener());
        checkBox2.addActionListener(new CheckBoxListener());
        checkBox3.addActionListener(new CheckBoxListener());
        checkBox4.addActionListener(new CheckBoxListener());
        checkBox5.addActionListener(new CheckBoxListener());

        add(checkBox1);
        add(checkBox2);
        add(checkBox3);
        add(checkBox4);
        add(checkBox5);
    }

    // ActionListener for radio buttons
    private class RadioButtonListener implements ActionListener {
        @Override
        public void actionPerformed(ActionEvent e) {
            displayField.setText("Radio: " + ((JRadioButton) e.getSource()).getText());
        }
    }

    // ActionListener for checkboxes
    private class CheckBoxListener implements ActionListener {
        @Override
        public void actionPerformed(ActionEvent e) {
            StringBuilder selected = new StringBuilder("Check: ");
            if (checkBox1.isSelected()) selected.append("1 ");
            if (checkBox2.isSelected()) selected.append("2 ");
            if (checkBox3.isSelected()) selected.append("3 ");
            if (checkBox4.isSelected()) selected.append("4 ");
            if (checkBox5.isSelected()) selected.append("5 ");
            displayField.setText(selected.toString().trim());
        }
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(() -> {
            JavaSwingRB frame = new JavaSwingRB();
            frame.setVisible(true);
        });
    }
}
